int symbol (int n, int k) {

   if (k == 0 || n == k) return 1;
   if (k>0 && n>0 && n>k) return symbol(n-1,k) + symbol(n-1, k-1);

}

double prawd (double n, double k, double p) {

        return pow(p,k)*pow(1-p,n-k)*symbol(n,k);
